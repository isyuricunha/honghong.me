## Usage

```tsx
import {
  Form,
  FormControl,
  FormDescription,
  FormField,
  FormItem,
  FormLabel,
  FormMessage
} from '@/components/ui/form'
```

```tsx
<Form {...form}>
  <form onSubmit={form.handleSubmit(onSubmit)}>
    <FormField
      control={form.control}
      name='...'
      render={({ field }) => (
        <FormItem>
          <FormLabel />
          <FormControl />
          <FormDescription />
          <FormMessage />
        </FormItem>
      )}
    />
  </form>
</Form>
```
