## Usage

```tsx title='app/layout.tsx'
import { Toaster } from '@tszhong0411/ui'

const Layout = ({ children }) => {
  return (
    <html>
      <body>
        {children}
        <Toaster />
      </body>
    </html>
  )
}

export default Layout
```
