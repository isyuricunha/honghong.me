## Usage

```tsx title='app/layout.tsx'
import { Toaster } from '@/components/ui/sonner'

const Layout = ({ children }) => {
  return (
    <html>
      <body>
        {children}
        <Toaster />
      </body>
    </html>
  )
}

export default Layout
```

```tsx
import { toast } from '@/components/ui/sonner'
```

