## Installation

```bash
pnpm dlx shadcn@latest add "https://docs.honghong.me/r/combobox"
```

## Usage

```tsx
import {
  Combobox,
  ComboboxContent,
  ComboboxControl,
  ComboboxInput,
  ComboboxItem,
  ComboboxItemGroup,
  ComboboxItemGroupLabel,
  ComboboxItemIndicator,
  ComboboxItemText,
  ComboboxTrigger
} from '@/components/ui/combobox'
```

```tsx
<Combobox collection={collection}>
  <ComboboxLabel>Framework</ComboboxLabel>
  <ComboboxTrigger>
    <ComboboxValue placeholder='Select a framework' />
  </ComboboxTrigger>
  <ComboboxContent>
    <ComboboxItemGroup>
      <ComboboxItemGroupLabel>Frameworks</ComboboxItemGroupLabel>
      {collection.map((item) => (
        <ComboboxItem key={item} item={item}>
          {item}
        </ComboboxItem>
      ))}
    </ComboboxItemGroup>
  </ComboboxContent>
</Combobox>
```
