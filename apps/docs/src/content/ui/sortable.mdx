---

<ComponentPreview name='sortable/sortable' />

## Usage

```tsx
import { Sortable, SortableContent, SortableItem, SortableOverlay } from '@tszhong0411/ui'
```

```tsx
<Sortable>
  <SortableContent>
    <SortableItem value='item-1'>Item 1</SortableItem>
    <SortableItem value='item-2'>Item 2</SortableItem>
    <SortableItem value='item-3'>Item 3</SortableItem>
  </SortableContent>
  <SortableOverlay />
</Sortable>
```
